Суть: Пользователь вводит химическую формулу (например, H²O или C⁶H¹²O⁶), а программа читает эту строку и выдает точную молярную массу.
Почему это круто:
Это учит работать со строками и структурами данных в C++.
Это демонстрация того, как вы понимаете структуру химических соединений.
Вы можете расширить его до поиска данных по таблице Менделеева (создать свой «словарь» элементов).

#include <iostream>
#include <string>
#include <map>

using namespace std;

int main() {
    // Наш справочник элементов
    map<string, double> masses = {
        {"H", 1.008}, {"C", 12.011}, {"O", 15.999}, {"N", 14.007}, {"S", 32.06}
    };

    string formula = "C⁶H¹²O⁶";
    double total_mass = 0;
    string last_element = "";

    for (int i = 0; i < formula.length(); i++) {
        char c = formula[i];

        // 1. Проверяем, является ли символ заглавной буквой
        if (c >= 'A' && c <= 'Z') {
            last_element = string(1, c);
            
            // Записываем результат: старая масса + новая масса
            total_mass = total_mass + masses[last_element];
        } 
        // 2. Проверяем, является ли символ цифрой
        else if (c >= '0' && c <= '9') {
            int count = c - '0';
            
            // Вычисляем добавку: масса * на количество (минус 1, так как мы уже добавили 1 атом)
            double extra_mass = masses[last_element] * (count - 1);
            
            // Записываем результат: старая масса + добавка
            total_mass = total_mass + extra_mass;
        }
    }

    cout << total_mass << endl;

    return 0;
}
